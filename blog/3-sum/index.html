LeetCode #15


Description

Given an array S of n integers, are there elements a, b, c in S such that a + b + c = 0? Find all unique triplets in the array which gives the sum of zero.


My Thoughts

Create a helper function twoSum(nums, target), to find if there are any two elements A and B in nums, A + B = target. Return all possible [A,B] pairs.

And in main function threeSum(nums), I just have to run a loop from index = 0 to nums.length. Call twoSum with target = -nums[index] first and push all possible three sum into answer array.

Solution

/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var threeSum = function(nums) {
    var index = 0;
    var ans = [];
    var check = {};
    nums.sort(function(a,b){
        return (a-b);
    });
    //console.log(nums);
    while(index < (nums.length)){
        if(nums[index] != nums[index-1]){
            var tmpNums = nums.slice(0);
            tmpNums.splice(index, 1);
            list = twoSum(tmpNums, -nums[index]);
            for(var i = 0; i < list.length; i++){
                list[i].push(nums[index]);
                list[i].sort(function(a,b){
                    return (a-b);
                });
                if(check[list[i]] != true){
                    ans.push(list[i]);
                }
                check[list[i]] = true;
            }
        }
        index++;
    }
    return ans;
};

var twoSum = function(nums, target){
    var list = [];
    var check = {};
    // nums.sort();
    // console.log(nums+" origin");
    var left = 0;
    var right = nums.length-1;
    while(left < right){
        if((nums[left] + nums[right]) == target){
            if(check[[nums[left], nums[right]]] != true){
                list.push([nums[left], nums[right]]);
            }
            check[[nums[left], nums[right]]] = true;
            left++;
            right--;
        }
        else if(nums[left] + nums[right] < target){
            left++;
        }
        else{
            right--;
        }
    }
    return list;
};
