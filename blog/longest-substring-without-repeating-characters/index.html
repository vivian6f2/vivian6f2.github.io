Longest Substring Without Repeating Characters

LeetCode #3


Description

Given a string, find the length of the longest substring without repeating characters.

Examples:

Given “abcabcbb”, the answer is “abc”, which the length is 3.

Given “bbbbb”, the answer is “b”, with the length of 1.

Given “pwwkew”, the answer is “wke”, with the length of 3. Note that the answer must be a substring, “pwke” is a subsequence and not a substring.

My Thoughts

Use a hash table to check if that character is in the substring. And use a queue to save the substring. Run loop from index = 0 to string.length, and there are two cases:

  • If character not in hash, push this character into queue and set hash[character] = true.
  • If not, then pop the queue until character not in hash anymore, also set hash[queue.pop()] = false.
var myFunc = function(s){
	var max_length = 0;
	for(index = 0; index < s.length; index++){
		if(s[index] not in hash){
			hash[s[index]] = true;
			queue.push(s[index]);
		}
		else{
			while((queue.length > 0) && (hash[s[index]] == true)){
				var tmp = queue.pop();
				hash[tmp] = false;
			}
			hash[s[index]] = true;
			queue.push(s[index]);
		}
		update max_length;
	}
	return max_length;
}


Solution

/**
 * @param {string} s
 * @return {number}
 */
var lengthOfLongestSubstring = function(s) {
    var queue = [];
    var hash = {};
    var index = 0;
    var maxLength = 0;
    while(index < s.length){
        if(hash[s[index]] != true){
            queue.push(s[index]);
            hash[s[index]] = true;
        }else{
            while((queue.length > 0) && (hash[s[index]] == true)){
                var tmp = queue.shift();
                hash[tmp] = false;
            }
            if(hash[s[index]] != true){
                hash[s[index]] = true;
                queue.push(s[index]);
            }
        }
        if(maxLength < queue.length){
            maxLength = queue.length;
        }
        index ++;
    }
    return maxLength;
};
