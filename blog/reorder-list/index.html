Reorder List

LeetCode #143


Description

Given a singly linked list L: L0?L1?…?Ln-1?Ln, reorder it to: L0?Ln?L1?Ln-1?L2?Ln-2?…

You must do this in-place without altering the nodes’ values.

For example, Given {1,2,3,4}, reorder it to {1,4,2,3}.

My Thoughts

I will divide this question into some smaller questions:

  1. Find the middle of the linked list. Just use two pointers, slow and fast. Slow pointer takes one step everytime while fast pointer takes two step. Until fast pointer reach the end of the linked list, slow pointer should be at the middle of linked list.

    var findMiddle = function(head){
    var slow = head;
    var fast = head;
    while(fast !== null && fast.next !== null){
        slow = slow.next;
        fast = fast.next.next;
    }
    return slow;
}

  2. Reverse second linked list (after middle). Also need two pointers, current and previous. For every node in the linked list, do the following steps:

    1. tmp = current.next, save current.next first.
    2. current.next = previous, link current node to previous node.
    3. previous = current, update previous node for next run.
    4. current = tmp, update current node for next run.
    var reverseList = function(head){
    var current = head;
    var previous = null;
    while(current){
        var tmp = current.next;
        current.next = previous;
        previous = current;
        current = tmp;
    }
    return previous;
}

  3. Merge two linked list. In this function, I would like to do following steps for every node in second linked list (take a->b, c->d as an example):

    1. tmp = secHead.next, save secHead.next first. (tmp = d)
    2. secHead.next = firstHead.next, move firstHead.next to secHead.next. (secHead = c->b)
    3. firstHead.next = secHead, move secHead back to firstHead.next. (firstHead = a->c->b)
    4. firstHead = firstHead.next.next, update firstHead for next run. (firstHead = b)
    5. secHead = tmp, update secHead for next run. (secHead = d)
    var merge = function(head, secHead){
    var firstHead = head;
    while(secHead != null){
        var tmp = secHead.next;
        secHead.next = firstHead.next;
        firstHead.next = secHead;
        if(firstHead.next.next){
            firstHead = firstHead.next.next;
        }else{
            firstHead = firstHead.next;
        }
        secHead = tmp;
    }
    return head;
}


Solution

/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {void} Do not return anything, modify head in-place instead.
 */
var reorderList = function(head) {
    if(head==null) return;
    if(head.next==null) return;
    
    var middle = findMiddle(head);
    var revSecHead = reverseList(middle);
    var tmp = head;
    var prev = null;

    // Take away the last node in first linked list (unnecessary node)
    while(tmp !== null){
        if(tmp.next == null){
            prev.next = null;
            tmp = tmp.next;
        }
        else{
            prev = tmp;
            tmp = tmp.next;
        }
    }

    head = merge(head, revSecHead);
    return;
    
};

var merge = function(head, secHead){
    var cur = head;
    while(secHead != null){
        var tmp = secHead.next;
        secHead.next = cur.next;
        cur.next = secHead;
        if(cur.next.next){
            cur = cur.next.next;
        }else{
            cur = cur.next;
        }
        secHead = tmp;
    }
    return head;
}

var findMiddle = function(head){
    var slow = head;
    var fast = head;
    while(fast !== null && fast.next !== null){
        slow = slow.next;
        fast = fast.next.next;
    }
    return slow;
}

var reverseList = function(head){
    var node = head;
    var previous = null;
    while(node){
        var tmp = node.next;
        node.next = previous;
        previous = node;
        node = tmp;
    }
    return previous;
}
